Directly inside

F5XC is capable of deploying and Customer Edge ( CE ) in the onprem or cloud environments.
This allows us to send the traffic to the application through an IPSEC tunnel and also provide internal and external load balancing and security services.

  1. Create a HTTP health check object 
    
                    
        
{
  "metadata": {
    "name": "http-hc",
    "labels": {},
    "annotations": {},
    "disable": false
  },
  "spec": {
    "http_health_check": {
      "host_header": "::makeid::.sales-public.f5demos.com",
      "path": "/",
      "use_http2": false,
      "headers": {}
    },
    "timeout": 3,
    "interval": 15,
    "unhealthy_threshold": 1,
    "healthy_threshold": 3,
    "jitter_percent": 30
  }
}
  1. Create the pool that will represent our backend services 
    
                    
        
{
  "metadata": {
    "name": "arcadia-onprem-origin-pool",
    "labels": {},
    "annotations": {},
    "disable": false
  },
  "spec": {
    "origin_servers": [
      {
        "private_ip": {
          "ip": "10.1.1.5",
          "site_locator": {
            "site": {
              "namespace": "system",
              "name": "::ceOnPrem::",
              "kind": "site"
            }
          },
          "outside_network": {}
        },
        "labels": {}
      },
      {
        "private_ip": {
          "ip": "10.1.1.6",
          "site_locator": {
            "site": {
              "namespace": "system",
              "name": "::ceOnPrem::",
              "kind": "site"
            }
          },
          "outside_network": {}
        },
        "labels": {}
      }
    ],
    "no_tls": {},
    "port": 31970,
    "same_as_endpoint_port": {},
    "healthcheck": [
      {
        "namespace": "::namespace::",
        "name": "http-hc",
        "kind": "healthcheck"
      }
    ],
    "loadbalancer_algorithm": "LB_OVERRIDE",
    "endpoint_selection": "LOCAL_PREFERRED"
  }
}
